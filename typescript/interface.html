    # 接口interface

    ts核心之一是结构的类型审查所以函数的参数就需要接口来描述从而统一结构类型， 我们可以吧接口当成一种类型来看即参数要符合该类型

    # 变量接口

    interface SquareConfig {
    color: string;//必填即必须的属性
    colour?: string;//选填非必须的
    readonly width?: number;//只读赋值声明之后不可修改
    readonly obj?:{a:number}//只读模式和变量声明一致
    [propName: string]: any;//任意数量类型参数
 }
 let a:SquareConfig={color:"red",width:12,obj:{a:1,b:2} as {a:number}}
 a.obj.a=2;
 console.log('a。obj.a: ', a.obj.a);//输出 2
 function createSquare  (config: SquareConfig){
      console.log('config: ', config); 
 }
 let mySquare = createSquare({ color: "red", width: 100 });

    # 类接口

    interface ClockInterface {
    currentTime: Date;
    setTime(d: Date);
}

class Clock implements ClockInterface {
    currentTime: Date;
    setTime(d: Date) {
        this.currentTime = d;
    }
    constructor(h: number, m: number) {}
}
//构造器接口
interface ClockConstructor {
     new (hour: number, minute: number): ClockInterface;
 }
 interface ClockInterface {
     show():any
     tick():any;
 }
 
 function createClock(ctor: ClockConstructor, hour: number, minute: number): ClockInterface {
     return new ctor(hour, minute);
 }

 class AnalogClock implements ClockInterface {
     private res:number;
     private hour:number;
     private minute:number;
     constructor(h: number, m: number) { 
         this.hour=h;
         this.minute=m;
          this.res=h+m;
     }
     show(){
          return this.res
     }
     tick() {
         console.log("tick tock");
     }
     end(){
         console.log("end")
     }
 }
 

 let analog = createClock(AnalogClock, 7, 32);
 analog.tick();
 analog.end();//error 报错 ClockInterface不存在end
 console.log('analog.res: ', analog.show());

    # 混合类型

    interface Counter {
     (start: number): string;
     interval: number;
     reset(): void;
 }
 
 function getCounter(){
     let counter = <Counter>function (start: number) { };
     counter.interval = 123;
     counter.reset = function () { };
     return counter;
 }
 
 let c = getCounter();
 c(10);
 c.reset();
 c.interval = 5.0;
