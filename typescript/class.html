    接触过C#，Java这类面向对象的语言的应该对类的构造函数和基本的继承原则都很熟悉。 下面有稍微复杂点的类继承，以及需要遵守的ts类继承规则。

    class hello{
     word:string;
     constructor(w:string){
          this.word=w;
     }
     show(){
          console.log("hello"+this.word)
     }
}
let hw=new hello("world")
class helloword extends hello{
     helloword:string
     constructor(w:string){
          //this.helloword=w+this.word   error 子类构造函数this操作必须在super()之后执行
          super(w)
          this.helloword=w+this.word   
     }
     show(){
          super.show();
          console.log("helloworld")
     }
     sonshow(){
         super.show();
     }
}
let hwson=new helloword("sonworld");
hwson.show(); 
// 输出
// hellosonworld
// helloworld
hwson.sonshow(); //输出 hellosonworld
console.log('hwson.helloword: ', hwson.helloword); // 输出 sonworldsonworld

    这里使用类一个super()方法，super可以理解为父类对象，如重写构造函数的时候必须要先执行父类构造函数即super()这是ts强制规定的，再比如重写show()方法中使用super.show()其实执行父类的show()方法。所以除了要注意super在构造函数中的用法在类定义的其他位置将super当成类定义中继承放父类部分即可。

    # protected 修饰符

    class Person {
    protected name: string;
    constructor(name: string) { this.name = name; }
}

class Employee extends Person {
    private department: string;

    constructor(name: string, department: string) {
        super(name)
        this.department = department;
    }

    public getElevatorPitch() {
        return `Hello, my name is ${this.name} and I work in ${this.department}.`;
    }
}

let howard = new Employee("Howard", "Sales");
console.log(howard.getElevatorPitch());

    # abstract 抽象类

    abstract class Department {

    constructor(public name: string) {
    }

    printName(): void {
        console.log('Department name: ' + this.name);
    }

    abstract printMeeting(): void; // 必须在派生类中实现
}

class AccountingDepartment extends Department {

    constructor() {
        super('Accounting and Auditing'); // 在派生类的构造函数中必须调用 super()
    }

    printMeeting(): void {
        console.log('The Accounting Department meets each Monday at 10am.');
    }

    generateReports(): void {
        console.log('Generating accounting reports...');
    }
}

let department: Department; // 允许创建一个对抽象类型的引用
department = new Department(); // 错误: 不能创建一个抽象类的实例
department = new AccountingDepartment(); // 允许对一个抽象子类进行实例化和赋值
department.printName();
department.printMeeting();
department.generateReports(); // 错误: 方法在声明的抽象类中不存在 与声明的时候的引用有关，如果去掉引用则不会报错引用可以看作是对结构的约束。
