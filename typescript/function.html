    # 函数

    ts的函数基本上和js的函数接近只是增加类类型声明，还附加了部分额外的功能方便使用。

    # 完整函数类型声明

    let myAdd: (x: number, y: number) => number =
    function(x: number, y: number): number { return x + y; };
//简写方式
function myAdd(x: number, y: number): number { return x + y; }
let myAdd =function(x: number, y: number): number { return x + y; };//function():number 的number其实也可以省略但是写返回值类型是个好习惯。

    # 可选参数，默认参数以及剩余参数

    function buildName(firstName: string, secondName="-", lastName?: string) {
    if (lastName)
        return firstName + secondName + lastName;
    else
        return firstName;
}
//剩余参数

function buildName(firstName: string, secondName="-", lastName?: string,...restOfName: string[]) {
     if (lastName)
         return firstName + secondName + lastName+restOfName.join("");
     else
         return firstName;
 }
 
let sname=buildName("aa",undefined,undefined,"dddd")
console.log('sname: ', sname);//输出 aa

    # this和箭头函数

    javascript中this是非常关键到一个特性，而javascript中this可以去understanding-javascript-function-invocation-and-this了解详细的内容。

    let deck = {
    suits: ["hearts", "spades", "clubs", "diamonds"],
    cards: Array(52),
    createCardPicker: function() {
        return function() {
            let pickedCard = Math.floor(Math.random() * 52);
            let pickedSuit = Math.floor(pickedCard / 13);

            return {suit: this.suits[pickedSuit], card: pickedCard % 13};
        }
    }
}
