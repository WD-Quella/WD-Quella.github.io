    # Set数据机构

    # 1.Set 常用用法

    Set相当于数组但是成员元素唯一不存在重复的值。

    let set=new Set([1,2,3,4,5,2])
//Set(5) {1, 2, 3, 4, 5}
set.size
//5
//添加元素
set.add(6)
//Set(6) {1, 2, 3, 4, 5,6}
//set存入节点 详单于对象存入set
const set = new Set(document.querySelectorAll('div'));
//等价于
const set = new Set();
document
 .querySelectorAll('div')
 .forEach(div => set.add(div));
//去重
[...set]
//[1,2,3,4,5]

    # 2.Set的操作方法

    Set.prototype.add(value)：添加某个值，返回 Set 结构本身。 Set.prototype.delete(value)：删除某个值，返回一个布尔值，表示删除是否成功。 Set.prototype.has(value)：返回一个布尔值，表示该值是否为Set的成员。 Set.prototype.clear()：清除所有成员，没有返回值。

    let set=new Set([1,2,3,4,5,2])
set.add(6)
//Set(6) {1, 2, 3, 4, 5, 6}
set.delete(2)
//Set(5) {1, 3, 4, 5, 6}
set.has(2)
//false
set.clear()
//Set(0) {}
//转化数组
set.add(1).add(2).add(3)
Array.from(set)
//[1,2,3]

    # 3.Set的遍历操作

    Set.prototype.keys()：返回键名的遍历器 Set.prototype.values()：返回键值的遍历器 Set.prototype.entries()：返回键值对的遍历器 Set.prototype.forEach()：使用回调函数遍历每个成员

    let set = new Set(['red', 'green', 'blue']);
for (let item of set.keys()) {
  console.log(item);
}
// red
// green
// blue
for (let item of set.values()) {
  console.log(item);
}
// red
// green
// blue
for (let item of set.entries()) {
  console.log(item);
}
// ["red", "red"]
// ["green", "green"]
// ["blue", "blue"]

    set的遍历运用

    let set = new Set([1, 2, 3]);
set = new Set([...set].map(x => x * 2));
// 返回Set结构：{2, 4, 6}

let set = new Set([1, 2, 3, 4, 5]);
set = new Set([...set].filter(x => (x % 2) == 0));
// 返回Set结构：{2, 4}

let a = new Set([1, 2, 3]);
let b = new Set([4, 3, 2]);

// 并集
let union = new Set([...a, ...b]);
// Set {1, 2, 3, 4}

// 交集
let intersect = new Set([...a].filter(x => b.has(x)));
// set {2, 3}

// （a 相对于 b 的）差集
let difference = new Set([...a].filter(x => !b.has(x)));
// Set {1}

    # 4.WeakSet

    weakset的成员只能是对象，也就意味着初始化比如为对象数组，不然会报错。 垃圾回收机制统计引用数量来执行内存清理，而weakset的引用输入弱引用不计入引用计数， 这也就导致了weakset不可遍历因为weakset的成员不稳定随时可能消失，而对这特性加以运用

    const foos = new WeakSet()
class Foo {
  constructor() {
    foos.add(this)
  }
  method () {
    if (!foos.has(this)) {
      throw new TypeError('Foo.prototype.method 只能在Foo的实例上调用！');
    }
  }
}

    # Map数据结构

    # 1.Map常用用法

    Map结构接近Object键值对（Hash结构），但是Map的键值可以是对象而没有只能是字符串的限制。

    let map=new Map([["one",1],["two",2]])
//Map(2) {"one" => 1, "two" => 2}
let map1=new Map([{1:1,0:"one"}])
//Map(1) {"one" => 1} map初始化本质取数组成员对象的0和1键的值。
map.size
//2
map.has("one")
//true
map.get("one")
//1
map.set("three",3)
//Map(3) {"one" => 1, "two" => 2, "three" => 3} 当使用对象作为键的时候要小心只有同一个键值才能取到对应的值。
map.set(-0, 123);
map.get(+0) // 123
map.set(true, 1);
map.set('true', 2);
map.get(true) // 1
map.set(undefined, 3);
map.get(undefined) // 3
map.set(null, 4);
map.get(null);//4
map.set(NaN, 123);
map.get(NaN) // 123
map.forEach((value,key)=>{console.log(key+":"+value)})
// one:1
// two:2
// three:3

    # 2.Map的操作方法

    基本同上Set方法的调用 Map.prototype.size Map.prototype.set(key, value) 返回Map对象所以可以链式调用。 Map.prototype.get(key) 获取key键的值 Map.prototype.delete(key) 删除某个值，返回一个布尔值，表示删除是否成功。 Map.prototype.has(key) 返回一个布尔值 Map.prototype.clear() 清除所有成员，没有返回值。

    # 3.Map的遍历

    Map.prototype.keys()：返回键名的遍历器。 Map.prototype.values()：返回键值的遍历器。 Map.prototype.entries()：返回所有成员的遍历器。 Map.prototype.forEach()：遍历 Map 的所有成员。

    const map = new Map([
  [1, 'one'],
  [2, 'two'],
  [3, 'three'],
]);
[...map.keys()]
// [1, 2, 3]
[...map.values()]
// ['one', 'two', 'three']
[...map.entries()]
// [[1,'one'], [2, 'two'], [3, 'three']]
[...map]
// [[1,'one'], [2, 'two'], [3, 'three']]
//Map本身没有filter和map方法，不过可以通过转化成数组来实现
const map0 = new Map()

const map1 = new Map(
  [...map0].filter(([k, v]) => k < 3)

const map2 = new Map(
  [...map0].map(([k, v]) => [k * 2, '_' + v])

    # 4.WeakMap

    同Map结构，但是WeakMap的键值必须是对象。 WeakMap键名所指向的对象(即键名对象)不计入垃圾回收机制。

    const wm = new WeakMap();
const element = document.getElementById('example');
wm.set(element, 'some information');
wm.get(element) // "some information"
//节点运用，当这个节点消失的时候weakmap会自动清除elemnet成员

let key = {};
let obj = {foo: 1};
wm.set(key, obj);
obj = null;
wm.get(key)
//weakmap 弱引用的只有键值，值所对应的对象引用不受影响。
